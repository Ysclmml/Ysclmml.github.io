这里开始从HelloWorld的demo开始阅读源码。

app = tornado.web.Application([
        (r'/', MainHandler),
    ])
app.listen(8888)
tornado.ioloop.IOLoop.current().start()

上面三行代码是先从实例化一个webApp开始的， 那么接下来开始看这个Application的源码

class Application(ReversibleRouter):
    r"""A collection of request handlers that make up a web application.

    Instances of this class are callable and can be passed directly to
    HTTPServer to serve the application::

        application = web.Application([
            (r"/", MainPageHandler),
        ])
        http_server = httpserver.HTTPServer(application)
        http_server.listen(8080)
        ioloop.IOLoop.current().start()
    """
        def __init__(
        self,
        handlers: Optional[_RuleList] = None,
        default_host: Optional[str] = None,
        transforms: Optional[List[Type["OutputTransform"]]] = None,
        **settings: Any
    ) -> None:
        """
        :param handlers: 处理业务逻辑的包含路由设置的handlers
        :param default_host: 默认的启动后的host
        :param transforms: 设置响应结果传输的方式
        :param settings:  application的配置
        """
        # 设置了transforms参数会在响应头中添加Content-Encoding='gzip'
        if transforms is None:
            self.transforms = []  # type: List[Type[OutputTransform]]
            # 如果需要压缩响应结果， 需要在结果中设置compress_response或gzip参数为True
            if settings.get("compress_response") or settings.get("gzip"):
                self.transforms.append(GZipContentEncoding)
        else:
            self.transforms = transforms
        # 默认host
        self.default_host = default_host
        # 加载所有初始化后传入的settings
        self.settings = settings
        # 模板相关的配置
        self.ui_modules = {
            "linkify": _linkify,
            "xsrf_form_html": _xsrf_form_html,
            "Template": TemplateModule,
        }
        self.ui_methods = {}  # type: Dict[str, Callable[..., str]]
        # 获取自定义的UIModule和UIMethods
        self._load_ui_modules(settings.get("ui_modules", {}))
        self._load_ui_methods(settings.get("ui_methods", {}))
        # 检查settings有没有配置静态资源的路径
        if self.settings.get("static_path"):
            path = self.settings["static_path"]
            handlers = list(handlers or [])
            # 默认的静态资源路径为/static/资源的地址
            static_url_prefix = settings.get("static_url_prefix", "/static/")
            # 静态资源的处理Handler， 可以自己传，也可以使用默认的Handler
            static_handler_class = settings.get(
                "static_handler_class", StaticFileHandler
            )
            # static_handler_args参数是传递给文件处理Handler执行initialize函数初始化的参数
            static_handler_args = settings.get("static_handler_args", {})
            static_handler_args["path"] = path
            # 默认会添加3条静态资源相关的路由地址
            for pattern in [
                re.escape(static_url_prefix) + r"(.*)",
                r"/(favicon\.ico)",
                r"/(robots\.txt)",
            ]:
                handlers.insert(0, (pattern, static_handler_class, static_handler_args))
        # 检查是否配置是debug模式
        if self.settings.get("debug"):
            # 自动重载
            self.settings.setdefault("autoreload", True)
            # 取消编译模板缓存
            self.settings.setdefault("compiled_template_cache", False)
            # 取消静态文件的hash值缓存
            self.settings.setdefault("static_hash_cache", False)
            # 提供处理跟踪信息
            self.settings.setdefault("serve_traceback", True)

        # 添加路由， 这个也是这个app最重要的一个部分
        self.wildcard_router = _ApplicationRouter(self, handlers)
        self.default_router = _ApplicationRouter(
            self, [Rule(AnyMatches(), self.wildcard_router)]
        )
        # 自动重载
        # Automatically reload modified modules
        if self.settings.get("autoreload"):
            from tornado import autoreload

            autoreload.start()

跟踪上面的_ApplicationRouter类代码

class _ApplicationRouter(ReversibleRuleRouter):
    """Routing implementation used internally by `Application`.
	这是一个用来绑定应用程序和RequestHandler的类， 通过路由建立桥梁。
    """

    def __init__(
        self, application: "Application", rules: Optional[_RuleList] = None
    ) -> None:
        assert isinstance(application, Application)
        self.application = application
        super().__init__(rules)

上面这个类继承自ReversibleRuleRouter， 继续跟踪代码往下。

class ReversibleRuleRouter(ReversibleRouter, RuleRouter):
    """A rule-based router that implements ``reverse_url`` method.
	这个类实现了反向解析类reverse_url方法， 这个功能一般是在模板里面使用的，能够根据名字反向解析出url
    """

    def __init__(self, rules: Optional[_RuleList] = None) -> None:
        self.named_rules = {}  # type: Dict[str, Any]
        super().__init__(rules)

    def process_rule(self, rule: "Rule") -> "Rule":
        rule = super().process_rule(rule)

        if rule.name:
            if rule.name in self.named_rules:
                app_log.warning(
                    "Multiple handlers named %s; replacing previous value", rule.name
                )
            # 将路由的名字与对应的rule规则映射起来。    
            self.named_rules[rule.name] = rule

        return rule

    def reverse_url(self, name: str, *args: Any) -> Optional[str]:
        """反向解析功能的实现（根据名字来获取映射的url）"""
        if name in self.named_rules:
            return self.named_rules[name].matcher.reverse(*args)

        for rule in self.rules:
            if isinstance(rule.target, ReversibleRouter):
                reversed_url = rule.target.reverse_url(name, *args)
                if reversed_url is not None:
                    return reversed_url

        return None

看到这里上面的这个类还是还是没有真正的添加路由规则， 它还继承了另一个核心类RuleRouter， 继续跟踪下去。 

class RuleRouter(Router):
    """Rule-based router implementation."""

    def __init__(self, rules: Optional[_RuleList] = None) -> None:
        """Constructs a router from an ordered list of rules::

            RuleRouter([
                Rule(PathMatches("/handler"), Target),
                # ... more rules
            ])

        You can also omit explicit `Rule` constructor and use tuples of arguments::

            RuleRouter([
                (PathMatches("/handler"), Target),
            ])

        `PathMatches` is a default matcher, so the example above can be simplified::

            RuleRouter([
                ("/handler", Target),
            ])

        In the examples above, ``Target`` can be a nested `Router` instance, an instance of
        `~.httputil.HTTPServerConnectionDelegate` or an old-style callable,
        accepting a request argument.

        :arg rules: a list of `Rule` instances or tuples of `Rule`
            constructor arguments.
        """
        self.rules = []  # type: List[Rule]
        if rules:
            # 这里是真正的添加规则
            self.add_rules(rules)

    def add_rules(self, rules: _RuleList) -> None:
        """Appends new rules to the router.

        :arg rules: a list of Rule instances (or tuples of arguments, which are
            passed to Rule constructor).
        """
        for rule in rules:
            if isinstance(rule, (tuple, list)):
                assert len(rule) in (2, 3, 4)
                if isinstance(rule[0], basestring_type):
                    rule = Rule(PathMatches(rule[0]), *rule[1:])
                else:
                    rule = Rule(*rule)

            self.rules.append(self.process_rule(rule))

看到这里其实已经豁然开朗了， 最终会把规则放入_ApplicationRouter对象的rules属性中， 就是app的wildcard_router属性中可以直接获取到。

接下来可以看看Rule对象的具体结构， 这个Rule对象保存了路由规则以及target处理的映射关系。

class Rule(object):
    """A routing rule."""

    def __init__(
        self,
        matcher: "Matcher",
        target: Any,
        target_kwargs: Optional[Dict[str, Any]] = None,
        name: Optional[str] = None,
    ) -> None:
        """Constructs a Rule instance.

        :arg Matcher matcher: a `Matcher` instance used for determining
            whether the rule should be considered a match for a specific
            request.
        :arg target: a Rule's target (typically a ``RequestHandler`` or
            `~.httputil.HTTPServerConnectionDelegate` subclass or even a nested `Router`,
            depending on routing implementation).
        :arg dict target_kwargs: a dict of parameters that can be useful
            at the moment of target instantiation (for example, ``status_code``
            for a ``RequestHandler`` subclass). They end up in
            ``target_params['target_kwargs']`` of `RuleRouter.get_target_delegate`
            method.
        :arg str name: the name of the rule that can be used to find it
            in `ReversibleRouter.reverse_url` implementation.
        """
        if isinstance(target, str):
            # import the Module and instantiate the class
            # Must be a fully qualified name (module.ClassName)
            target = import_object(target)

        self.matcher = matcher  # type: Matcher
        self.target = target
        self.target_kwargs = target_kwargs if target_kwargs else {}
        self.name = name

一条路由规则组成结构还是比较简单的， 一共就4部分， 上面的英文文档其实已经说的比较清楚了

  • matcher 其实就是在我们写的正则路由中包装了一层
  • target request请求开始后路由到对应的target对象去处理， 这个target对象可以是典型的RequestHandler或是httputil.HTTPServerConnectionDelegate的实现类或是还可以是一个嵌套的Router类实例
  • target_kwargs 顾名思义， 就是给target（请求处理对象）的参数
  • name 路由反向解析的名字

到这也基本可以明白一个app初始化后， 主要就是一些内部参数的初始化， 以及最重要的路由绑定等等处理。
